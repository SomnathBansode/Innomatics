ATM Withdrawal System

                function atmWithdrawal(balance, amount, pin, enteredPin) {
                        if (enteredPin !== pin) return "Incorrect PIN";
                        if (amount > balance) return "Insufficient Funds";
                        if (amount % 100 !== 0) return "Enter amount in multiples of 100";
                        return `Withdrawal successful. Remaining balance: $${balance - amount}`;
                }
                
                
Online Shopping Discount & Free Shipping

                function calculateFinalAmount(orderAmount) {
                        let discount = orderAmount > 1000 ? 0.20 : orderAmount >= 500 ? 0.10 : 0;
                        let finalAmount = orderAmount - (orderAmount * discount);
                        let shipping = finalAmount > 50 ? 0 : 10;
                        return finalAmount + shipping;
                }
                
                
Student Grading System with Extra Credit

                function calculateGrade(marks, attendance) {
                        if (attendance > 90) marks += 5;
                        return marks >= 90 ? "A" : marks >= 80 ? "B" : marks >= 70 ? "C" : marks >= 60 ? "D" : "F";
                }
                
                
Smart Traffic Light System

                function trafficLightControl(density) {
                        const times = { "Heavy Traffic": 60, "Moderate Traffic": 40, "Light Traffic": 20 };
                        return times[density] || "Invalid traffic density";
                }
                
                
Movie Ticket Pricing with Time and Age Discount

                function calculateTicketPrice(age, showTime) {
                        let price = 12;
                        if (showTime < 17) price *= 0.8;
                        if (age > 60) price *= 0.7;
                        if (age < 12) price *= 0.6;
                        return price;
                }
                
                
Job Application Filter

                function isEligibleForJob(age, experience, qualification) {
                        return age >= 21 && age <= 55 && experience >= 2 && qualification === "Bachelor's Degree";
                }
                
                
E-commerce Coupon Redemption

                function applyCoupon(orderAmount, couponCode) {
                        let discount = couponCode === "DISCOUNT10" && orderAmount > 500 ? 0.10 : 0;
                        let freeShipping = couponCode === "FREESHIP" && orderAmount > 200;
                        let finalAmount = orderAmount - (orderAmount * discount);
                        let shipping = freeShipping || finalAmount > 50 ? 0 : 10;
                        return finalAmount + shipping;
                }
                
                
Fitness Membership Plan

                function choosePlan(planType, wantsTrainer, wantsDietPlan) {
                        return planType === "VIP" || (wantsTrainer && wantsDietPlan) ? "VIP Plan - $80/month" : 
                                     planType === "Premium" || wantsTrainer ? "Premium Plan - $50/month" : 
                                     "Basic Plan - $20/month";
                }
                
                
Electricity Bill Calculation with Peak Hours

                function calculateElectricityBill(units, timeOfDay) {
                        let rate = units > 300 ? 3 : units > 100 ? 4 : 5;
                        if (timeOfDay >= 20 || timeOfDay < 8) rate *= 1.1;
                        return units * rate;
                }
                
                
Flight Ticket Booking System

                function calculateFlightFare(classType, luggageWeight, isStudent, isSenior) {
                        let fare = 300 + (classType === "Business" ? 200 : classType === "First" ? 500 : 0);
                        fare += Math.max(0, Math.floor((luggageWeight - 20) / 10)) * 50;
                        let discount = isStudent ? 0.15 : isSenior ? 0.10 : 0;
                        return fare * (1 - discount);
                }
